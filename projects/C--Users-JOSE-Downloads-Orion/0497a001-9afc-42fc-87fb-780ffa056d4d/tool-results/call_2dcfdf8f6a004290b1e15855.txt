   113→// Read stdin with timeout
   114→async function readStdin(timeout = 3000): Promise<string> {
   115→  return new Promise((resolve) => {
   116→    let data = '';
   117→    const timer = setTimeout(() => resolve(data), timeout);
   118→    process.stdin.on('data', chunk => { data += chunk.toString(); });
   119→    process.stdin.on('end', () => { clearTimeout(timer); resolve(data); });
   120→    process.stdin.on('error', () => { clearTimeout(timer); resolve(''); });
   121→  });
   122→}
   123→
   124→// Classify prompt using AI inference
   125→async function classifyPrompt(prompt: string): Promise<{
   126→  depth: 'FULL' | 'ITERATION' | 'MINIMAL';
   127→  capabilities: string[];
   128→  skills: string[];
   129→  thinking: string[];
   130→}> {
   131→  const result = await inference({
   132→    systemPrompt: CLASSIFICATION_SYSTEM_PROMPT,
   133→    userPrompt: prompt,
   134→    level: 'standard',
   135→    expectJson: true,
   136→    timeout: 10000, // 10s — if Sonnet can't classify in 10s, fall back to FULL
   137→  });
   138→
   139→  if (result.success && result.parsed) {
   140→    const parsed = result.parsed as { depth?: string; capabilities?: string[]; skills?: string[]; thinking?: string[] };
   141→    const depth = ['FULL', 'ITERATION', 'MINIMAL'].includes(parsed.depth || '')
   142→      ? (parsed.depth as 'FULL' | 'ITERATION' | 'MINIMAL')
   143→      : 'FULL';
   144→    const capabilities = Array.isArray(parsed.capabilities)
   145→      ? parsed.capabilities.filter((c: string) => c in CAPABILITY_MAP)
   146→      : [];
   147→    const skills = Array.isArray(parsed.skills)
   148→      ? parsed.skills.filter((s: string) => typeof s === 'string' && s.length > 0)
   149→      : [];
   150→    const thinking = Array.isArray(parsed.thinking)
   151→      ? parsed.thinking.filter((t: string) => t in THINKING_MAP)
   152→      : [];
   153→    return { depth, capabilities, skills, thinking };
   154→  }
   155→
   156→  // Inference failed — safe default: FULL, no specific capabilities
   157→  return { depth: 'FULL', capabilities: [], skills: [], thinking: [] };
   158→}

   160→// Build the reminder output
   161→function buildReminder(
   162→  depth: 'FULL' | 'ITERATION' | 'MINIMAL',
   163→  capabilities: string[],
   164→  skills: string[],
   165→  thinking: string[],
   166→): string {
   167→  const capabilitySection = capabilities.length > 0
   168→    ? `\n⚡ DETECTED CAPABILITIES (based on your request):\n${capabilities.map(c => {
   169→        const cap = CAPABILITY_MAP[c];
   170→        return cap ? `• ${cap.name} → ${cap.agents}` : '';
   171→      }).filter(Boolean).join('\n')}\n\nYou SHOULD spawn these agents in BUILD/EXECUTE phases.`
   172→    : '';
   173→
   174→  const skillSection = skills.length > 0
   175→    ? `\n🎯 DETECTED SKILLS (Pass 1 hints — validate in THINK against ISC):\n${skills.map(s => `• ${s}`).join('\n')}`
   176→    : '';
   177→
   178→  const thinkingSection = thinking.length > 0
   179→    ? `\n🧠 SUGGESTED THINKING TOOLS (Pass 1 hints — run justify-exclusion in THINK):\n${thinking.map(t => {
   180→        const tool = THINKING_MAP[t];
   181→        return tool ? `• ${tool.name} — ${tool.description}` : '';
   182→      }).filter(Boolean).join('\n')}`
   183→    : '';
   184→
   185→  switch (depth) {
   186→    case 'FULL':
   187→      return `<system-reminder>
   188→ALGORITHM REQUIRED — DEPTH: FULL
   189→Nothing escapes the Algorithm. Your response MUST use the 7-phase format:
   190→- Start with: 🤖 PAI ALGORITHM header
   191→- Include ALL phases: OBSERVE → THINK → PLAN → BUILD → EXECUTE → VERIFY → LEARN
   192→- Use TaskCreate for ISC criteria, TaskList to display them
   193→- End with voice line
   194→${capabilitySection}${skillSection}${thinkingSection}
   195→</system-reminder>`;
   196→
   197→    case 'ITERATION':
   198→      return `<system-reminder>
   199→ALGORITHM REQUIRED — DEPTH: ITERATION
   200→Nothing escapes the Algorithm. Use condensed format:
   201→🤖 PAI ALGORITHM ═════
   202→🔄 ITERATION on: [context]
   203→🔧 CHANGE: [what's different]
   204→✅ VERIFY: [evidence]
   205→🗣️ ${getDAName()}: [result]
   206→${capabilitySection}${skillSection}${thinkingSection}
   207→</system-reminder>`;
   208→
   209→    case 'MINIMAL':
   210→      return `<system-reminder>
   211→ALGORITHM REQUIRED — DEPTH: MINIMAL
   212→Nothing escapes the Algorithm. Use header format:
   213→🤖 PAI ALGORITHM (v0.2.23) ═════
   214→   Task: [6 words]
   215→📋 SUMMARY: [what was done]
   216→🗣️ ${getDAName()}: [voice line]
   217→</system-reminder>`;
   218→  }
   219→}

   221→async function main() {
   222→  try {

   226→      process.exit(0);
   227→    }
   228→
   229→    const input = await readStdin();
   230→    if (!input) {
   231→      process.exit(0);
   232→    }
   233→
   234→    const data = JSON.parse(input);
   235→    const prompt = data.prompt || data.user_prompt || '';
   236→
   237→    if (!prompt) {
   238→      process.exit(0);
   239→    }
   240→
   241→    // AI-powered classification — no regex, no keywords, no length heuristics
   242→    const { depth, capabilities, skills, thinking } = await classifyPrompt(prompt);
   243→    const reminder = buildReminder(depth, capabilities, skills, thinking);
   244→
   245→    console.log(reminder);
   246→    process.exit(0);
   247→  } catch (err) {
   248→    // On any error, output FULL as safe default
   249→    console.log(`<system-reminder>
   250→ALGORITHM REQUIRED — DEPTH: FULL
   251→Nothing escapes the Algorithm. Your response MUST use the 7-phase format:
   252→- Start with: 🤖 PAI ALGORITHM header
   253→- Include ALL phases: OBSERVE → THINK → PLAN → BUILD → EXECUTE → VERIFY → LEARN
   254→- Use TaskCreate for ISC criteria, TaskList to display them
   255→- End with voice line
   256→</system-reminder>`);
   257→    process.exit(0);
   258→  }
   259→}
